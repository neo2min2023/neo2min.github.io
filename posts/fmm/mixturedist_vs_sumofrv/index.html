pdf of weighted sum of normal pdfs VS pdf of weighted sum of normally distributed random variables.

xs <- seq(-6, 10, by=0.01)
sigma1 <- 1
sigma2 <- 2
mu1 <- -1
mu2 <- 3
pdf1 <- 1/sqrt((2*pi*sigma1^2)) * exp(-1/(2*sigma1^2) * (xs - mu1)^2)
pdf2 <- 1/sqrt((2*pi*sigma2^2)) * exp(-1/(2*sigma2^2) * (xs - mu2)^2)

omega1 <- 0.2
omega2 <- 0.8
pdf1_weighted <- omega1 * pdf1
pdf2_weighted <- omega2 * pdf2
pdf_mixed <- pdf1_weighted + pdf2_weighted
mu_sum <- omega1 * mu1 + omega2 * mu2
sigma_sum <- sqrt(omega1^2 * sigma1^2 + omega2^2 * sigma2^2)
pdf_sum <- 1/sqrt((2*pi*sigma_sum^2)) * exp(-1/(2*sigma_sum^2) * (xs - mu_sum)^2)

sampleDf <- data.table(x1=rnorm(1e5, mean=mu1, sd=sigma1), x2=rnorm(1e5, mean=mu2, sd=sigma2),grp=sample(0:1, 1e5, replace=TRUE, prob=c(omega1, omega2)), x=0)
sampleDf[, x:=fifelse(grp==0, x1, x2)]
sampleDf[, xSum:=omega1*x1+omega2*x2]

x1_kde <- density(sampleDf[, x1])
x2_kde <- density(sampleDf[, x2])
x12_kde <- sampleDf[, density(c(x1, x2), weights=c(rep(omega1*1/.N, .N), rep(omega2*1/.N, .N)))]
xSum_kde <- density(sampleDf[, xSum])

#png("./mixtrueDistributionVSsumOfRVs.png", width=3500, height=2500, res=200)
par(mfrow=c(1,2))
plot(xs, pdf1, type="l", col="red1", lty=2, lwd=2, main="theoretic distributions")
lines(xs, pdf2, col="cyan", lty=2, lwd=2)
lines(xs, pdf1_weighted, col="red2", lwd=2)
lines(xs, pdf2_weighted, col="cyan", lwd=2)
lines(xs, pdf_mixed, col="purple", lwd=3)
polygon(xs, pdf_mixed, col = rgb(0, 0, 1, alpha = 0.1))
lines(xs, pdf_sum, lwd=2, col = rgb(0, 1, 0, alpha = 0.9))
polygon(xs, pdf_sum, col = rgb(0, 1, 0, alpha = 0.1))
legend("topright", inset=.05, title="distributions", c(expression(y==-frac(1, sqrt(2*pi*sigma[1]^2))*e^(frac(1, 2*sigma[1]^2)*(x-mu[1])^2)), expression(x2 %~% N(mu[2], sigma[2]^2)), expression(omega[1]*pdf[1]), expression(omega[2]*pdf[2]), expression(omega[1]*pdf[1]+omega[2]*pdf[2]), expression(pdf(omega[1]*x[1]+omega[2]*x[2]))), lty=c(2, 2, 1, 1, 1, 1), col=c("red1", "cyan", "red2", "cyan", "purple", "green"))
